18    pub(crate) id_delta: u16,
19    pub(crate) id_range_offset: u16,

22/// Segment mapping to delta values (format 4) subtable of the `cmap` table.
23#[derive(Debug, Clone)]
24pub(crate) struct SegmentDeltas<'a> {
25    pub(crate) segments: Vec<SegmentWithDelta>,
26    pub(crate) glyph_id_array: &'a [u8],
27}
28
29impl<'a> SegmentDeltas<'a> {
30    fn parse(mut cursor: Cursor<'a>) -> Result<Self, ParseError> {
31        cursor.read_u16_checked(|format| {
32            if format != 4 {
33                return Err(ParseErrorKind::UnexpectedTableFormat(format));
34            }
35            Ok(())
36        })?;
37
38        let remaining_len = cursor.read_u16_checked(|subtable_len| {
39            Ok(subtable_len
40                .checked_sub(4)
41                .ok_or(ParseErrorKind::UnexpectedEof)? as usize)
42        })?;
43        cursor = cursor.range(0..remaining_len)?;
44
45        cursor.skip(2)?; // language
46        let segment_count = cursor.read_u16()? / 2;
47        cursor.skip(6)?; // searchRange, entrySelector, rangeShift
48
49        let vec_len = 2 * usize::from(segment_count);
50        let mut end_codes = cursor.split_at(vec_len)?;
51        cursor.skip(2)?; // reserved padding
52        let mut start_codes = cursor.split_at(vec_len)?;
53        let mut id_deltas = cursor.split_at(vec_len)?;
54        let mut id_range_offsets = cursor.split_at(vec_len)?;
55
56        let segments = (0..segment_count).map(|_| {
57            Ok(SegmentWithDelta {
58                start_code: start_codes.read_u16()?,
59                end_code: end_codes.read_u16()?,
60                id_delta: id_deltas.read_u16()?,
61                id_range_offset: id_range_offsets.read_u16()?,
62            })
63        });
64
65        Ok(Self {
66            segments: segments.collect::<Result<_, ParseError>>()?,
67            glyph_id_array: cursor.bytes,
68        })
69    }
70
71    fn map_char(&self, c: char) -> Result<u16, ParseError> {
72        let Ok(c) = u16::try_from(c as u32) else {
73            return Ok(0); // missing glyph
74        };
75
76        let segment_idx = self
77            .segments
78            .binary_search_by_key(&c, |segment| segment.end_code)
79            .unwrap_or_else(|pos| pos);
80        let segment = &self.segments[segment_idx];
81        if segment.start_code > c {
82            return Ok(0); // missing glyph
83        }
84
85        if segment.id_range_offset == 0 {
86            Ok(segment.id_delta.wrapping_add(c))
87        } else {
88            // Offset is counted from the start of `idRangeOffsets`
89            let mut byte_offset = 2 * segment_idx;
90            byte_offset += usize::from(segment.id_range_offset);
91            byte_offset += 2 * usize::from(c - segment.start_code);
92
93            if byte_offset < 2 * self.segments.len() {
94                return Err(ParseError {
95                    kind: ParseErrorKind::OffsetOutOfBounds(byte_offset),
96                    offset: 0,
97                    table: Some(TableTag::CMAP),
98                });
99            }
100            // Shift the offset to count from the start of `glyphIdArray`
101            byte_offset -= 2 * self.segments.len();
102            let glyph_id_bytes = self
103                .glyph_id_array
104                .get(byte_offset..(byte_offset + 2))
105                .ok_or(ParseError {
106                    kind: ParseErrorKind::OffsetOutOfBounds(byte_offset),
107                    offset: 0,
108                    table: Some(TableTag::CMAP),
109                })?;
110            let glyph_id = u16::from_be_bytes(glyph_id_bytes.try_into().unwrap());
111            Ok(segment.id_delta.wrapping_add(glyph_id))
112        }
113    }
114}

116#[derive(Debug, Clone, Copy)]
117pub(crate) struct SequentialMapGroup {
118    pub(crate) start_char_code: u32,
119    pub(crate) end_char_code: u32,
120    pub(crate) start_glyph_id: u32,
121}
122
123impl SequentialMapGroup {
124    pub(crate) fn map_unchecked(&self, ch: char) -> u32 {
125        u32::from(ch) - self.start_char_code + self.start_glyph_id
126    }
127}

129/// Segmented coverage (format 12) subtable of the `cmap` table.
130#[derive(Debug, Default, Clone)]
131pub(crate) struct SegmentedCoverage {
132    pub(crate) groups: Vec<SequentialMapGroup>,
133}
134
135impl SegmentedCoverage {
136    fn parse(mut cursor: Cursor<'_>) -> Result<Self, ParseError> {
137        cursor.read_u16_checked(|format| {
138            if format != 12 {
139                return Err(ParseErrorKind::UnexpectedTableFormat(format));
140            }
141            Ok(())
142        })?;
143
144        cursor.skip(2)?; // reserved
145
146        let remaining_len = cursor.read_u32_checked(|subtable_len| {
147            Ok(subtable_len
148                .checked_sub(8)
149                .ok_or(ParseErrorKind::UnexpectedEof)? as usize)
150        })?;
151        cursor = cursor.range(0..remaining_len)?;
152
153        cursor.skip(4)?; // language
154        let num_groups = cursor.read_u32()?;
155        let groups = (0..num_groups).map(|_| {
156            Ok(SequentialMapGroup {
157                start_char_code: cursor.read_u32()?,
158                end_char_code: cursor.read_u32()?,
159                start_glyph_id: cursor.read_u32()?,
160            })
161        });
162
163        Ok(Self {
164            groups: groups.collect::<Result<_, ParseError>>()?,
165        })
166    }
167
168    fn map_char(&self, ch: char) -> u16 {
169        let ch = u32::from(ch);
170        let group_idx = self
171            .groups
172            .binary_search_by_key(&ch, |group| group.end_char_code)
173            .unwrap_or_else(|pos| pos);
174        let Some(group) = self.groups.get(group_idx) else {
175            return 0; // `ch` exceeds `end_char_code` for the last segment
176        };
177        if group.start_char_code > ch {
178            return 0; // missing glyph
179        }
180        let glyph_id = ch - group.start_char_code + group.start_glyph_id;
181        glyph_id.try_into().expect("glyph ID exceeds u16::MAX")
182    }
183}

185#[derive(Debug, Clone)]
186pub(crate) enum CmapTable<'a> {
187    Deltas(SegmentDeltas<'a>),
188    Coverage(SegmentedCoverage),
189}
190
191impl<'a> CmapTable<'a> {
192    pub(crate) const UNICODE_PLATFORM: u16 = 0;
193    const WINDOWS_PLATFORM: u16 = 3;
194
195    pub(super) fn parse(mut cursor: Cursor<'a>) -> Result<Self, ParseError> {
196        let table_cursor = cursor;
197        cursor.read_u16_checked(|version| {
198            if version != 0 {
199                return Err(ParseErrorKind::UnexpectedTableVersion(version.into()));
200            }
201            Ok(())
202        })?;
203
204        let num_tables = cursor.read_u16()?;
205        let mut this = None;
206        for _ in 0..num_tables {
207            let platform_id = cursor.read_u16()?;
208            let encoding_id = cursor.read_u16()?;
209            let offset = cursor.read_u32()?;
210            let expected_table_format = match (platform_id, encoding_id) {
211                (Self::UNICODE_PLATFORM, 3) | (Self::WINDOWS_PLATFORM, 1) => {
212                    CmapTableFormat::SegmentDeltas
213                }
214                (Self::UNICODE_PLATFORM, 4) | (Self::WINDOWS_PLATFORM, 10) => {
215                    CmapTableFormat::SegmentedCoverage
216                }
217                _ => continue, // unsupported table format
218            };
219
220            match expected_table_format {
221                CmapTableFormat::SegmentDeltas if this.is_none() => {
222                    let mut subtable = table_cursor;
223                    subtable.skip(offset as usize)?;
224                    this = Some(Self::Deltas(SegmentDeltas::parse(subtable)?));
225                }
226                CmapTableFormat::SegmentedCoverage if this.is_none() => {
227                    let mut subtable = table_cursor;
228                    subtable.skip(offset as usize)?;
229                    this = Some(Self::Coverage(SegmentedCoverage::parse(subtable)?));
230                }
231                _ => { /* We've already got a necessary table; do nothing */ }
232            }
233        }
234
235        this.ok_or_else(|| cursor.err(ParseErrorKind::NoSupportedCmap))
236    }
237
238    pub(super) fn map_char(&self, ch: char) -> Result<u16, ParseError> {
239        match self {
240            Self::Deltas(deltas) => deltas.map_char(ch),
241            Self::Coverage(coverage) => Ok(coverage.map_char(ch)),
242        }
243    }
