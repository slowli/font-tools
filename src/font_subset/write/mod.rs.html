25fn uint_base128_len(val: u32) -> usize {
26    if val == 0 {
27        1
28    } else {
29        val.ilog2() as usize / 7 + 1
30    }
31}
32
33#[allow(clippy::cast_possible_truncation)] // intentional
34fn write_uint_base128(buffer: &mut Vec<u8>, val: u32) {
35    if val >= 1 << 28 {
36        buffer.push(0x80 | (val >> 28) as u8);
37    }
38    if val >= 1 << 21 {
39        buffer.push(0x80 | (val >> 21) as u8);
40    }
41    if val >= 1 << 14 {
42        buffer.push(0x80 | (val >> 14) as u8);
43    }
44    if val >= 1 << 7 {
45        buffer.push(0x80 | (val >> 7) as u8);
46    }
47    buffer.push((val & 127) as u8);
48}

50impl CmapTable<'static> {
51    fn from_map(map: &[(char, u16)]) -> Self {
52        let coverage = Self::create_coverage(map);
53        let can_be_encoded_as_deltas = map
54            .last()
55            .is_none_or(|&(ch, _)| u32::from(ch) < u32::from(u16::MAX));
56        if can_be_encoded_as_deltas {
57            #[allow(clippy::cast_possible_truncation)]
58            // `_ as u16` is safe due to the `can_be_encoded_as_deltas` check
59            let delta_segments = coverage.groups.iter().map(|group| {
60                let start_code = group.start_char_code as u16;
61                SegmentWithDelta {
62                    start_code,
63                    end_code: group.end_char_code as u16,
64                    id_delta: (group.start_glyph_id as u16).wrapping_sub(start_code),
65                    id_range_offset: 0,
66                }
67            });
68            // Add en empty segment with `start_code == end_code == 0xffff` as per spec.
69            let delta_segments = delta_segments.chain([SegmentWithDelta {
70                start_code: u16::MAX,
71                end_code: u16::MAX,
72                id_delta: 1, // will map `start_code` to glyph #0 (the missing glyph) as recommended
73                id_range_offset: 0,
74            }]);
75            Self::Deltas(SegmentDeltas {
76                segments: delta_segments.collect(),
77                glyph_id_array: &[],
78            })
79        } else {
80            Self::Coverage(coverage)
81        }
82    }
83
84    fn create_coverage(map: &[(char, u16)]) -> SegmentedCoverage {
85        let mut groups = vec![];
86        let [(first_char, first_idx), rest @ ..] = map else {
87            return SegmentedCoverage::default();
88        };
89        let mut current_group = SequentialMapGroup {
90            start_char_code: (*first_char).into(),
91            end_char_code: (*first_char).into(),
92            start_glyph_id: (*first_idx).into(),
93        };
94
95        for &(ch, glyph_idx) in rest {
96            if u32::from(ch) == current_group.end_char_code + 1
97                && u32::from(glyph_idx) == current_group.map_unchecked(ch)
98            {
99                current_group.end_char_code += 1;
100            } else {
101                let prev_group = mem::replace(
102                    &mut current_group,
103                    SequentialMapGroup {
104                        start_char_code: ch.into(),
105                        end_char_code: ch.into(),
106                        start_glyph_id: glyph_idx.into(),
107                    },
108                );
109                groups.push(prev_group);
110            }
111        }
112
113        groups.push(current_group);
114        SegmentedCoverage { groups }
115    }
116}

118impl CmapTable<'_> {
119    fn write(&self, writer: &mut Vec<u8>) {
120        write_u16(writer, 0); // table version
121        write_u16(writer, 1); // num_tables
122
123        write_u16(writer, CmapTable::UNICODE_PLATFORM);
124        let encoding_id = match self {
125            Self::Deltas(_) => 3,
126            Self::Coverage(_) => 4,
127        };
128        write_u16(writer, encoding_id);
129        write_u32(writer, 12); // subtable_offset
130
131        match self {
132            Self::Deltas(deltas) => deltas.write(writer),
133            Self::Coverage(coverage) => coverage.write(writer),
134        }
135    }
136}

138impl SegmentDeltas<'_> {
139    fn subtable_len(&self) -> usize {
140        16 + 8 * self.segments.len()
141    }
142
143    fn write(&self, writer: &mut Vec<u8>) {
144        write_u16(writer, 4); // subtable format
145        write_u16(
146            writer,
147            self.subtable_len()
148                .try_into()
149                .expect("subtable_len overflow"),
150        );
151        write_u16(writer, 0); // language
152
153        let segment_count = u16::try_from(self.segments.len()).expect("segments.len() overflow");
154        write_u16(writer, 2 * segment_count);
155        let entry_selector = u16::try_from(segment_count.ilog2()).unwrap();
156        let search_range = 1 << (entry_selector + 1);
157        write_u16(writer, search_range);
158        write_u16(writer, entry_selector);
159        let range_shift = 2 * segment_count - search_range;
160        write_u16(writer, range_shift);
161
162        for segment in &self.segments {
163            write_u16(writer, segment.end_code);
164        }
165        write_u16(writer, 0); // reserved padding
166        for segment in &self.segments {
167            write_u16(writer, segment.start_code);
168        }
169        for segment in &self.segments {
170            write_u16(writer, segment.id_delta);
171        }
172        for segment in &self.segments {
173            write_u16(writer, segment.id_range_offset);
174        }
175        writer.extend_from_slice(self.glyph_id_array);
176    }
177}
178
179impl SegmentedCoverage {
180    fn subtable_len(&self) -> usize {
181        16 + 12 * self.groups.len()
182    }
183
184    fn write(&self, writer: &mut Vec<u8>) {
185        write_u16(writer, 12); // subtable format
186        write_u16(writer, 0); // reserved
187
188        write_u32(
189            writer,
190            self.subtable_len()
191                .try_into()
192                .expect("subtable_len overflow"),
193        );
194        write_u32(writer, 0); // language
195        write_u32(
196            writer,
197            self.groups.len().try_into().expect("groups.len() overflow"),
198        );
199        for group in &self.groups {
200            write_u32(writer, group.start_char_code);
201            write_u32(writer, group.end_char_code);
202            write_u32(writer, group.start_glyph_id);
203        }
204    }

207impl FontSubset<'_> {
208    /// Serializes this subset to the OpenType format.
209    pub fn to_opentype(&self) -> Vec<u8> {
210        self.to_writer().into_opentype()
211    }
212
213    /// Serializes this subset to the WOFF2 format.
214    pub fn to_woff2(&self) -> Vec<u8> {
215        self.to_writer().into_woff2()
216    }
217
218    fn to_writer(&self) -> FontWriter {
219        let cmap = CmapTable::from_map(&self.char_map);
220
221        let mut writer = FontWriter::default();
222        writer.write_table(TableTag::CMAP, |buffer| cmap.write(buffer));
223        if let Some(cvt) = self.font.cvt {
224            writer.write_raw_table(TableTag::CVT, cvt.as_ref());
225        }
226        if let Some(fpgm) = self.font.fpgm {
227            writer.write_raw_table(TableTag::FPGM, fpgm.as_ref());
228        }
229
230        let number_of_h_metrics = writer.write_table(TableTag::HMTX, |buffer| {
231            HmtxTable::write_for_glyphs(&self.glyphs, buffer)
232        });
233        let mut hhea = self.font.hhea;
234        hhea.number_of_h_metrics = number_of_h_metrics;
235        writer.write_table(TableTag::HHEA, |buffer| {
236            hhea.write(buffer);
237        });
238
239        let maxp = self.font.maxp.as_ref();
240        writer.write_table(TableTag::MAXP, |buffer| {
241            // Patch the number of glyphs (u16 at bytes 4..6), and leave other bytes intact.
242            buffer.extend_from_slice(&maxp[..4]);
243            // `unwrap()` should be safe: the subset shouldn't contain >65536 glyphs because the original font doesn't.
244            write_u16(buffer, self.glyphs.len().try_into().unwrap());
245            buffer.extend_from_slice(&maxp[6..]);
246        });
247
248        // TODO: reduce `name` table?
249        writer.write_raw_table(TableTag::NAME, self.font.name.as_ref());
250        writer.write_raw_table(TableTag::OS2, self.font.os2.as_ref());
251
252        let post = self.font.post.as_ref();
253        writer.write_table(TableTag::POST, |buffer| {
254            // Truncate the `post` table to not contain glyph names
255            write_u32(buffer, 0x_00030000); // version
256            buffer.extend_from_slice(&post[4..32]);
257        });
258
259        if let Some(prep) = self.font.prep {
260            writer.write_raw_table(TableTag::PREP, prep.as_ref());
261        }
262
263        let locations = writer.write_table(TableTag::GLYF, |buffer| {
264            let mut locations = vec![0];
265            let initial_offset = buffer.len();
266            for glyph in &self.glyphs {
267                let glyph = &glyph.inner;
268                glyph.write(buffer);
269                locations.push(buffer.len() - initial_offset);
270            }
271            locations
272        });
273
274        let loca_format = writer.write_table(TableTag::LOCA, |buffer| {
275            LocaTable::write(&locations, buffer)
276        });
277        writer.write_table(TableTag::HEAD, |buffer| {
278            Self::write_head_table(self.font.head.as_ref(), loca_format, buffer);
279        });
280
281        writer
282    }
283
284    fn write_head_table(original: &[u8], loca_format: LocaFormat, writer: &mut Vec<u8>) {
285        const LOCA_FORMAT_OFFSET: usize = 50;
286
287        writer.extend_from_slice(&original[..Font::HEAD_CHECKSUM_OFFSET]);
288        write_u32(writer, 0); // Zero the checksum as per spec. It will be adjusted later
289        writer.extend_from_slice(&original[Font::HEAD_CHECKSUM_OFFSET + 4..LOCA_FORMAT_OFFSET]);
290        write_u16(
291            writer,
292            match loca_format {
293                LocaFormat::Short => 0,
294                LocaFormat::Long => 1,
295            },
296        );
297        writer.extend_from_slice(&original[LOCA_FORMAT_OFFSET + 2..]);
298    }
299}

301impl HmtxTable<'_> {
302    fn write_for_glyphs(glyphs: &[GlyphWithMetrics<'_>], writer: &mut Vec<u8>) -> u16 {
303        let mut number_of_h_metrics = glyphs.len();
304        while let Some([prev, current]) = glyphs[..number_of_h_metrics].last_chunk::<2>() {
305            if prev.advance != current.advance {
306                break;
307            }
308            number_of_h_metrics -= 1;
309        }
310
311        for (i, glyph) in glyphs.iter().enumerate() {
312            if i < number_of_h_metrics {
313                write_u16(writer, glyph.advance);
314                write_u16(writer, glyph.lsb);
315            } else {
316                write_u16(writer, glyph.lsb);
317            }
318        }
319
320        // `unwrap()` should be safe: `number_of_h_metrics` <= number of glyphs, which doesn't exceed u16::MAX
321        number_of_h_metrics.try_into().unwrap()
322    }
323}

332impl LocaTable<'_> {
333    fn write(locations: &[usize], writer: &mut Vec<u8>) -> LocaFormat {
334        let all_even = locations.iter().all(|&loc| loc % 2 == 0);
335        let in_bounds = locations
336            .last()
337            .is_none_or(|&loc| loc <= usize::from(u16::MAX) * 2);
338        if all_even && in_bounds {
339            for &loc in locations {
340                #[allow(clippy::cast_possible_truncation)]
341                // doesn't happen due to the preceding check
342                write_u16(writer, (loc / 2) as u16);
343            }
344            LocaFormat::Short
345        } else {
346            for &loc in locations {
347                write_u32(writer, u32::try_from(loc).expect("glyph location overflow"));
348            }
349            LocaFormat::Long
350        }
351    }
352}

354#[derive(Debug, Clone, Copy)]
355#[cfg_attr(test, derive(PartialEq))]
356struct TableRecord {
357    tag: TableTag,
358    checksum: u32,
359    /// Offset is initially recorded relative to the table data start. It's always 4-byte aligned.
360    offset: u32,
361    length: u32,
362}
363
364impl TableRecord {
365    const BYTE_LEN: usize = 16;
366
367    fn write_opentype(&self, writer: &mut Vec<u8>) {
368        writer.extend_from_slice(&self.tag.0);
369        write_u32(writer, self.checksum);
370        write_u32(writer, self.offset);
371        write_u32(writer, self.length);
372    }
373
374    fn self_checksum(&self) -> u32 {
375        u32::from_be_bytes(self.tag.0)
376            .wrapping_add(self.checksum)
377            .wrapping_add(self.offset)
378            .wrapping_add(self.length)
379    }
380
381    fn woff2_len(&self) -> usize {
382        1 /* flags */ + uint_base128_len(self.length)
383    }
384
385    fn write_woff2(&self, buffer: &mut Vec<u8>) {
386        const NULL_TRANSFORM: u8 = 0b_1100_0000;
387
388        let flags = match self.tag {
389            TableTag::CMAP => 0,
390            TableTag::HEAD => 1,
391            TableTag::HHEA => 2,
392            TableTag::HMTX => 3,
393            TableTag::MAXP => 4,
394            TableTag::NAME => 5,
395            TableTag::OS2 => 6,
396            TableTag::POST => 7,
397            TableTag::CVT => 8,
398            TableTag::FPGM => 9,
399            TableTag::GLYF => 10 | NULL_TRANSFORM,
400            TableTag::LOCA => 11 | NULL_TRANSFORM,
401            TableTag::PREP => 12,
402            _ => unreachable!("subsetting only produces well-known tables"),
403        };
404        buffer.push(flags);
405        write_uint_base128(buffer, self.length);
406    }
407}

409#[derive(Debug, Clone, Default)]
410struct FontWriter {
411    tables: Vec<TableRecord>,
412    /// Contains *aligned* table data
413    table_data: Vec<u8>,
414}
415
416impl FontWriter {
417    const SFNT_HEADER_LEN: usize = 12;
418    const WOFF2_HEADER_LEN: usize = 48;
419
420    fn write_table<T>(&mut self, tag: TableTag, with: impl FnOnce(&mut Vec<u8>) -> T) -> T {
421        let offset = self.table_data.len();
422        debug_assert_eq!(offset % 4, 0, "unaligned offset: {offset}");
423
424        let output = with(&mut self.table_data);
425        let length = self.table_data.len() - offset;
426        // Pad the table heap to a 4-byte boundary.
427        if length % 4 > 0 {
428            let zero_padding = 4 - length % 4;
429            self.table_data.extend(iter::repeat_n(0_u8, zero_padding));
430        }
431
432        let checksum = Font::checksum(&self.table_data[offset..]);
433        self.tables.push(TableRecord {
434            tag,
435            checksum,
436            offset: u32::try_from(offset).expect("table offset overflow"),
437            length: u32::try_from(length).expect("table length overflow"),
438        });
439        output
440    }
441
442    fn write_raw_table(&mut self, tag: TableTag, content: &[u8]) {
443        self.write_table(tag, |buffer| buffer.extend_from_slice(content));
444    }
445
446    fn write_sfnt_header(&self) -> Vec<u8> {
447        let mut buffer = vec![];
448        write_u32(&mut buffer, Font::SFNT_VERSION);
449
450        // `unwrap()`s are safe: we don't have many tables written.
451        let table_count = u16::try_from(self.tables.len()).unwrap();
452        write_u16(&mut buffer, table_count);
453        let entry_selector = u16::try_from(table_count.ilog2()).unwrap();
454        let search_range = 1 << (4 + entry_selector);
455        write_u16(&mut buffer, search_range);
456        write_u16(&mut buffer, entry_selector);
457        let range_shift = 16 * table_count - search_range;
458        write_u16(&mut buffer, range_shift);
459
460        debug_assert_eq!(buffer.len(), Self::SFNT_HEADER_LEN);
461        buffer
462    }
463
464    /// Returns the starting offset of table data.
465    fn data_offset(&self) -> usize {
466        Self::SFNT_HEADER_LEN + self.tables.len() * TableRecord::BYTE_LEN
467    }
468
469    fn into_opentype(mut self) -> Vec<u8> {
470        let mut buffer = self.write_sfnt_header();
471        self.adjust_data(Font::checksum(&buffer));
472
473        self.tables.sort_unstable_by_key(|record| record.tag.0);
474        for record in &self.tables {
475            record.write_opentype(&mut buffer);
476        }
477        buffer.extend(self.table_data);
478        buffer
479    }
480
481    fn adjust_data(&mut self, sfnt_header_checksum: u32) {
482        let data_offset = self.data_offset();
483        let data_offset_u32 = u32::try_from(data_offset).expect("data_offset overflow");
484
485        let mut file_checksum = sfnt_header_checksum;
486        for record in &mut self.tables {
487            record.offset += data_offset_u32;
488            file_checksum = file_checksum
489                .wrapping_add(record.self_checksum())
490                .wrapping_add(record.checksum);
491        }
492        self.patch_head_table(file_checksum, data_offset);
493    }
494
495    fn checksum_adjustment_offset(&self) -> usize {
496        let head_table = self
497            .tables
498            .iter()
499            .find(|record| record.tag == TableTag::HEAD)
500            .expect("head table is always present");
501        head_table.offset as usize + Font::HEAD_CHECKSUM_OFFSET
502    }
503
504    fn patch_head_table(&mut self, file_checksum: u32, data_offset: usize) {
505        let checksum_adjustment = Font::SFNT_CHECKSUM.wrapping_sub(file_checksum);
506
507        // At this point, the table offset already includes the heap offset, so we need to subtract it.
508        let offset = self.checksum_adjustment_offset() - data_offset;
509        self.table_data[offset..offset + 4].copy_from_slice(&checksum_adjustment.to_be_bytes());
510    }
511
512    fn into_woff2(mut self) -> Vec<u8> {
513        const WOFF2_SIGNATURE: u32 = 0x_774f_4632;
514
515        self.adjust_data(Font::checksum(&self.write_sfnt_header()));
516
517        let compressed_data = self.compress_data();
518        let tables_len = self
519            .tables
520            .iter()
521            .map(TableRecord::woff2_len)
522            .sum::<usize>();
523        let mut file_len = Self::WOFF2_HEADER_LEN + tables_len + compressed_data.len();
524        if file_len % 4 != 0 {
525            file_len += 4 - file_len % 4;
526        }
527
528        let mut buffer = vec![];
529        write_u32(&mut buffer, WOFF2_SIGNATURE);
530        write_u32(&mut buffer, Font::SFNT_VERSION);
531        write_u32(
532            &mut buffer,
533            file_len.try_into().expect("file length overflow"),
534        );
535        // `unwrap()` is safe: we don't write many tables
536        write_u16(&mut buffer, self.tables.len().try_into().unwrap());
537        write_u16(&mut buffer, 0); // reserved
538
539        let decompressed_len = self.data_offset() + self.table_data.len();
540        // `unwrap`s are safe, since `file_len` fits into u32.
541        write_u32(&mut buffer, decompressed_len.try_into().unwrap());
542        write_u32(&mut buffer, compressed_data.len().try_into().unwrap());
543        write_u32(&mut buffer, 0); // WOFF version
544        write_u32(&mut buffer, 0); // metadata offset
545        write_u32(&mut buffer, 0); // metadata length
546        write_u32(&mut buffer, 0); // original metadata length
547        write_u32(&mut buffer, 0); // private block offset
548        write_u32(&mut buffer, 0); // private block length
549        debug_assert_eq!(buffer.len(), Self::WOFF2_HEADER_LEN);
550
551        for record in &self.tables {
552            record.write_woff2(&mut buffer);
553        }
554        debug_assert_eq!(buffer.len(), Self::WOFF2_HEADER_LEN + tables_len);
555        buffer.extend(compressed_data);
556
557        // Pad `buffer` to be 4-byte aligned. This is required even though we don't have metadata or private blocks.
558        if buffer.len() % 4 != 0 {
559            let padding = 4 - buffer.len() % 4;
560            buffer.extend(iter::repeat_n(0, padding));
561        }
562        debug_assert_eq!(file_len, buffer.len());
563        buffer
564    }
565}

567impl Glyph<'_> {
568    fn write(&self, writer: &mut Vec<u8>) {
569        match self {
570            Self::Empty => { /* do nothing */ }
571            Self::Simple(bytes) => {
572                writer.extend_from_slice(bytes);
573            }
574            Self::Composite {
575                header,
576                components,
577                instructions,
578            } => {
579                write_u16(writer, u16::MAX); // numberOfContours = -1
580                writer.extend_from_slice(header);
581                for component in components {
582                    component.write(writer);
583                }
584                writer.extend_from_slice(instructions);
585            }
586        }
587    }
588}

590impl GlyphComponent {
591    fn write(&self, writer: &mut Vec<u8>) {
592        write_u16(writer, self.flags);
593        write_u16(writer, self.glyph_idx);
594        match self.args {
595            GlyphComponentArgs::U16(args) => write_u16(writer, args),
596            GlyphComponentArgs::U32(args) => write_u32(writer, args),
597        }
598        match self.transform {
599            TransformData::None => { /* do nothing */ }
600            TransformData::Scale(val) => write_u16(writer, val),
601            TransformData::TwoScales([x, y]) => {
602                write_u16(writer, x);
603                write_u16(writer, y);
604            }
605            TransformData::Affine([xx, xy, yx, yy]) => {
606                write_u16(writer, xx);
607                write_u16(writer, xy);
608                write_u16(writer, yx);
609                write_u16(writer, yy);
610            }
611        }
612    }
