8struct TableDataReader<'a> {
9    writer: &'a FontWriter,
10    data_offset: u32,
11    table_idx: usize,
12    pos_in_table: usize,
13}
14
15impl<'a> TableDataReader<'a> {
16    fn new(writer: &'a FontWriter) -> Self {
17        debug_assert!(
18            writer.tables.windows(2).all(|window| {
19                let [prev, next] = window else {
20                    unreachable!();
21                };
22                prev.offset + prev.length <= next.offset
23            }),
24            "table records need to be ordered by offsets"
25        );
26        let data_offset = writer.tables.first().map_or(0, |record| record.offset);
27
28        Self {
29            writer,
30            data_offset,
31            table_idx: 0,
32            pos_in_table: 0,
33        }
34    }
35
36    fn read_chunk<'data>(
37        &self,
38        range: ops::Range<usize>,
39        data: &'data mut [u8],
40    ) -> (usize, Option<&'data mut [u8]>) {
41        let remaining = &self.writer.table_data[range];
42        if remaining.len() < data.len() {
43            let (head, tail) = data.split_at_mut(remaining.len());
44            head.copy_from_slice(remaining);
45            // Continue reading from the next table
46            (remaining.len(), Some(tail))
47        } else {
48            data.copy_from_slice(&remaining[..data.len()]);
49            (data.len(), None)
50        }
51    }
52}
53
54impl brotli::CustomRead<()> for TableDataReader<'_> {
55    fn read(&mut self, mut data: &mut [u8]) -> Result<usize, ()> {
56        let mut total_read = 0;
57        loop {
58            let Some(table) = self.writer.tables.get(self.table_idx) else {
59                return Ok(total_read); // nothing left to read
60            };
61
62            let adjusted_offset = (table.offset - self.data_offset) as usize;
63            let start_offset = adjusted_offset + self.pos_in_table;
64            let end_offset = adjusted_offset + table.length as usize;
65            let (read, remaining_data) = self.read_chunk(start_offset..end_offset, data);
66            total_read += read;
67
68            if let Some(remaining_data) = remaining_data {
69                // Move to the next table
70                self.table_idx += 1;
71                self.pos_in_table = 0;
72                data = remaining_data;
73            } else {
74                // Run out of the output buffer
75                self.pos_in_table += read;
76                debug_assert!(self.pos_in_table <= table.length as usize);
77                return Ok(total_read);
78            }
79        }
80    }
81}

83#[derive(Default)]
84struct Buffer(Vec<u8>);
85
86impl brotli::CustomWrite<()> for Buffer {
87    fn write(&mut self, data: &[u8]) -> Result<usize, ()> {
88        self.0.extend_from_slice(data);
89        Ok(data.len())
90    }
91
92    fn flush(&mut self) -> Result<(), ()> {
93        Ok(())
94    }

139impl FontWriter {
140    pub(super) fn compress_data(&self) -> Vec<u8> {
141        let mut buffer = Buffer::default();
142        ::brotli::BrotliCompressCustomIo(
143            &mut TableDataReader::new(self),
144            &mut buffer,
145            &mut [0_u8; 4_096],
146            &mut [0_u8; 4_096],
147            &::brotli::enc::BrotliEncoderParams::default(),
148            GlobalAlloc,
149            &mut |_, _, _, _| { /* do nothing */ },
150            (),
151        )
152        .expect("Writing to Vec never fails");
153
154        buffer.0
155    }
156}

158#[cfg(test)]
159mod tests {
160    use std::fs;
161
162    use brotli::CustomRead;
163    use test_casing::test_casing;
164
165    use super::*;
166    use crate::{Font, FontSubset};
167
168    #[test_casing(5, [1, 10, 100, 1000, 100_000])]
169    fn table_data_reader_works_as_expected(chunk_size: usize) {
170        let font_bytes = fs::read("examples/FiraMono-Regular.ttf").unwrap();
171        let font = Font::new(&font_bytes).unwrap();
172        let chars = (' '..='~').collect();
173        let subset = FontSubset::new(font, &chars).unwrap();
174        let writer = subset.to_writer();
175
176        let mut data_reader = TableDataReader::new(&writer);
177        let mut buffer = vec![0; 100_000];
178
179        let read = buffer
180            .chunks_mut(chunk_size)
181            .map(|chunk| data_reader.read(chunk).unwrap())
182            .sum::<usize>();
183        let expected_read = writer
184            .tables
185            .iter()
186            .map(|record| record.length as usize)
187            .sum::<usize>();
188        assert_eq!(read, expected_read);
189
190        let mut pos = 0;
191        for record in &writer.tables {
192            let offset = record.offset as usize;
193            let len = record.length as usize;
194            assert_eq!(
195                writer.table_data[offset..offset + len],
196                buffer[pos..pos + len]
197            );
198            pos += len;
199        }
200    }
